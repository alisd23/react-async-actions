import React from 'react';
import { renderToString } from 'react-dom/server';
import { asyncActions, fireAsyncActions } from '../index';
 
const ACTION_DELAY = 50;
 
let store;
 
const asyncAction = (func) =>
  new Promise(resolve => {
    setTimeout(() => {
      this.addFruit(fruit);
      resolve();
    }, ACTION_DELAY)
  });
 
const createComponent = (actionsToFire) => {
  const Component = () => <div />;
  return asyncActions(actionsToFire)(Component);
};
 
beforeEach(() => {
  store = {
    addFruit: jest.fn(),
    addFruitAsync: (fruit) => asyncAction(() => this.addFruit(fruit))
  };
});
 
describe('server rendering', () => {
  it('fireAsyncActions waits til promises have resolved', () => {
    const Component = createComponent(() => [store.addFruitAsync('orange')])
    const element = <Component />;
    fireAsyncActions(element)
      .then(() => {
        console.log('ORANGE');
        expect(true).toBe(false);
        expect(store.addFruit.mock.calls[0]).toBe('orange');
      });
 
    expect(store.addFruit.mock.calls.length).toBe(0);
  })
});
